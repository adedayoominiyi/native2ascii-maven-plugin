17  package org.codehaus.mojo.native2ascii.mojo;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.util.Iterator;
22  
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugins.annotations.LifecyclePhase;
25  import org.apache.maven.plugins.annotations.Mojo;
26  import org.apache.maven.plugins.annotations.Parameter;
27  import org.codehaus.mojo.native2ascii.Native2Ascii;
28  import org.codehaus.plexus.util.FileUtils;
29  
30  /**
31   * Converts files with characters in any supported character encoding to one with ASCII and/or
32   * Unicode escapes.
33   * <p>
34   * This mojo converts files inplace, it is not recommended to execute it on src subdirectories.
35   */
36  @Mojo(name = "inplace", defaultPhase = LifecyclePhase.PROCESS_RESOURCES)
37  public class Native2AsciiInplaceMojo extends AbstractNative2AsciiMojo {
38  
39    /**
40     * Both source and target directory.
41     */
42    @Parameter(required = true, defaultValue = "${native2ascii.dir}")
43    protected File dir;
44  
45  
46    @Override
47    protected File getSourceDirectory() {
48      return dir;
49    }
50  
51  
52    @Override
53    public void executeTransformation(final Iterator<File> files) throws MojoExecutionException {
54      while (files.hasNext()) {
55        File file = files.next();
56        getLog().info("Processing " + file.getAbsolutePath());
57        try {
58          // Convert file in-place
59          File tempFile = File.createTempFile(file.getName(), "native2ascii");
60          new Native2Ascii(getLog()).nativeToAscii(file, tempFile, encoding);
61          FileUtils.rename(tempFile, file);
62          getLog().info("File converted successfuly: " + file.getAbsolutePath());
63        } catch (IOException e) {
64          throw new MojoExecutionException("Unable to convert " + file.getAbsolutePath(), e);
65        }
66      }
67    }
68  }
