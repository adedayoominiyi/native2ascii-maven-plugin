17  package org.codehaus.mojo.native2ascii;
18  
19  import static org.junit.Assert.assertEquals;
20  
21  import org.apache.maven.plugin.logging.SystemStreamLog;
22  import org.junit.Test;
23  
24  public class Native2AsciiTest {
25  
26    final Native2Ascii native2Ascii = new Native2Ascii(new SystemStreamLog());
27  
28  
29    @Test
30    public void test() {
31      assertEquals(null, native2Ascii.nativeToAscii(null));
32      assertEquals("~", native2Ascii.nativeToAscii("~"));
33      assertEquals("\\u007F", native2Ascii.nativeToAscii(""));
34      // chinese character is random since I don't understand it
35      assertEquals("\\u7C97", native2Ascii.nativeToAscii("粗"));
36      assertEquals("\\u0158\\u00EDze\\u010Dek ut\\u00EDk\\u00E1 a \\u0159e\\u017Ee zat\\u00E1\\u010Dky! \\u00A7",
37          native2Ascii.nativeToAscii("Řízeček utíká a řeže zatáčky! §"));
38    }
39  
40  }
