17  package org.codehaus.mojo.native2ascii.mojo;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertNull;
21  
22  import java.io.File;
23  import java.io.FileInputStream;
24  import java.util.Properties;
25  
26  import org.codehaus.mojo.native2ascii.mojo.Native2AsciiMojo;
27  import org.junit.Test;
28  
29  
30  /**
31   * @author David Matějček
32   */
33  public class Native2AsciiMojoTest {
34  
35    @Test
36    public void testFile() throws Exception {
37      Native2AsciiMojo mojo = new Native2AsciiMojo();
38      mojo.encoding = "UTF-8";
39      mojo.srcDir = new File(Native2AsciiMojoTest.class.getClassLoader().getResource("xxx.properties").toURI()).getParentFile();
40      mojo.targetDir = mojo.srcDir.getParentFile();
41      mojo.includes = new String[] {"xxx.properties"};
42      mojo.excludes = new String[0];
43      mojo.execute();
44  
45      Properties properties = new Properties();
46      final FileInputStream inputStream = new FileInputStream(new File(mojo.targetDir, "xxx.properties"));
47      try {
48        properties.load(inputStream);
49      } finally {
50        inputStream.close();
51      }
52      assertEquals("~", properties.get("1"));
53      assertEquals("", properties.get("2"));
54      assertNull(properties.get("3"));
55      assertEquals("粗 Řízeček utíká a řeže zatáčky! §", properties.get("4"));
56      assertNull(properties.get("5"));
57      assertEquals("more tests needed!", properties.get("6"));
58      assertEquals("", properties.get("7"));
59    }

61  }
