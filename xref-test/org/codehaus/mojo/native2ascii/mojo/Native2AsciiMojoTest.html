17  package org.codehaus.mojo.native2ascii.mojo;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertNull;
21  import static org.junit.Assert.assertTrue;
22  
23  import java.io.File;
24  import java.io.FileInputStream;
25  import java.io.IOException;
26  import java.net.URISyntaxException;
27  import java.util.Properties;
28  
29  import org.junit.Test;
30  
31  /**
32   * @author David Matějček
33   */
34  public class Native2AsciiMojoTest {
35  
36    @Test
37    public void testFile() throws Exception {
38      final Native2AsciiMojo mojo = new Native2AsciiMojo();
39      mojo.encoding = "UTF-8";
40      mojo.srcDir = getSourceDirectory();
41      mojo.targetDir = mojo.srcDir.getParentFile();
42      mojo.includes = new String[] {"xxx.properties"};
43      mojo.excludes = new String[0];
44      mojo.execute();
45  
46      final Properties properties = loadFile(new File(mojo.targetDir, "xxx.properties"));
47      assertEquals("~", properties.get("1"));
48      assertEquals("", properties.get("2"));
49      assertNull(properties.get("3"));
50      assertEquals("粗 Řízeček utíká a řeže zatáčky! §", properties.get("4"));
51      assertNull(properties.get("5"));
52      assertEquals("more tests needed!", properties.get("6"));
53      assertEquals("", properties.get("7"));
54    }
55  
56  
57    @Test
58    public void testSubdirectories() throws Exception {
59      final Native2AsciiMojo mojo = new Native2AsciiMojo();
60      mojo.encoding = "UTF-8";
61      mojo.srcDir = getSourceDirectory();
62      mojo.targetDir = mojo.srcDir.getParentFile();
63      mojo.includes = new String[] {"subdirs/**/*"};
64      mojo.excludes = new String[0];
65      mojo.execute();
66  
67      final File file = new File(mojo.targetDir.getCanonicalPath() + File.separator + "subdirs/x2/x3/sub.properties");
68      assertTrue("file does not exist: " + file, file.exists());
69      final Properties properties = loadFile(file);
70      assertEquals("šílené!", properties.get("crazy"));
71    }
72  
73  
74    private File getSourceDirectory() throws URISyntaxException {
75      return new File(Native2AsciiMojoTest.class.getClassLoader().getResource("xxx.properties").toURI()).getParentFile();
76    }
77  
78  
79    private Properties loadFile(final File file) throws IOException {
80      final Properties properties = new Properties();
81      final FileInputStream inputStream = new FileInputStream(file);
82      try {
83        properties.load(inputStream);
84        return properties;
85      } finally {
86        inputStream.close();
87      }
88    }
89  }
